## `[controlled]: boolean{:ts}`

export const Controlled = () => {
  const [value, setValue] = useState('')
  return (
    <TextField
      variant="filled"
      name="controlled"
      onChange={(e) => setValue(e.target.value)}
      value={value}
      message={`value: ${value}`}
    />
  )
}

<Demo>
  <TextField variant="filled" name="uncontrolled" defaultValue="default" />
  <Controlled />
</Demo>

```tsx
const [value, setValue] = useState('')

<TextField
  variant="filled"
  name="uncontrolled"
  defaultValue="default"
/>

<TextField
  variant="filled"
  name="controlled"
  onChange={(e) => setValue(e.target.value)}
  value={value}

/>
```
